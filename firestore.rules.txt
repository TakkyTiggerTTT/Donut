rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function profilePath(userId) {
      return /databases/$(database)/documents/users/$(userId);
    }

    function settingsPath() {
      return /databases/$(database)/documents/settings/site;
    }

    function hasProfile() {
      return signedIn() && exists(profilePath(request.auth.uid));
    }

    function profile() {
      return get(profilePath(request.auth.uid)).data;
    }

    function isOwner() {
      return hasProfile() && profile().role == "owner";
    }

    function isAdmin() {
      return hasProfile() && (profile().role == "admin" || profile().role == "owner");
    }

    function siteWip() {
      return exists(settingsPath()) && get(settingsPath()).data.get("wipEnabled", false) == true;
    }

    function siteOpen() {
      return !siteWip() || isOwner();
    }

    function saveListBanned() {
      return hasProfile() && profile().get("savelistBanned", false) == true;
    }

    function canUseSaveList() {
      return siteOpen() && hasProfile() && !saveListBanned();
    }

    function canDismissOwnPromotion(userId) {
      return signedIn() &&
        request.auth.uid == userId &&
        resource.data.role == "student" &&
        request.resource.data.role == resource.data.role &&
        request.resource.data.classId == resource.data.classId &&
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(["promotion", "promotionSeenAt"]) &&
        request.resource.data.promotion.diff(resource.data.promotion).affectedKeys()
          .hasOnly(["pending"]) &&
        resource.data.promotion.pending == true &&
        request.resource.data.promotion.pending == false;
    }

    function validSaveImage(image) {
      return image is string &&
        image.size() <= 330000 &&
        image.matches('^data:image/(jpeg|png|webp);base64,[A-Za-z0-9+/=]+$');
    }

    function validSaveImages(images) {
      return images is list &&
        images.size() <= 2 &&
        (
          images.size() == 0 ||
          (images.size() == 1 && validSaveImage(images[0])) ||
          (images.size() == 2 && validSaveImage(images[0]) && validSaveImage(images[1]))
        );
    }

    function validSaveFields() {
      return request.resource.data.title is string &&
        request.resource.data.title.size() > 0 &&
        request.resource.data.title.size() <= 120 &&
        request.resource.data.bodyHtml is string &&
        request.resource.data.bodyHtml.size() <= 20000 &&
        validSaveImages(request.resource.data.images);
    }

    function validNewSave() {
      return request.resource.data.keys().hasOnly([
          "ownerId", "ownerName", "ownerRole", "ownerClassId", "ownerClassName",
          "title", "bodyHtml", "images", "createdAt", "updatedAt"
        ]) &&
        request.resource.data.ownerId == request.auth.uid &&
        request.resource.data.ownerRole == profile().role &&
        request.resource.data.ownerName == profile().get("name", profile().get("username", "User")) &&
        request.resource.data.ownerClassId == profile().get("classId", "") &&
        validSaveFields();
    }

    function validOwnSaveUpdate() {
      return resource.data.ownerId == request.auth.uid &&
        request.resource.data.ownerId == resource.data.ownerId &&
        request.resource.data.ownerName == resource.data.ownerName &&
        request.resource.data.ownerRole == resource.data.ownerRole &&
        request.resource.data.ownerClassId == resource.data.ownerClassId &&
        request.resource.data.ownerClassName == resource.data.ownerClassName &&
        request.resource.data.createdAt == resource.data.createdAt &&
        request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(["title", "bodyHtml", "images", "updatedAt"]) &&
        validSaveFields();
    }

    match /settings/{settingId} {
      allow read: if settingId == "site";
      allow create, update, delete: if settingId == "site" && isOwner();
    }

    match /users/{userId} {
      allow read: if signedIn() && (isAdmin() || request.auth.uid == userId);

      allow create: if isOwner() ||
        (isAdmin() && request.resource.data.role == "student");

      allow update: if isOwner() ||
        (
          isAdmin() &&
          resource.data.role == "student" &&
          request.resource.data.role == "student"
        ) || canDismissOwnPromotion(userId);

      allow delete: if isOwner() ||
        (isAdmin() && resource.data.role == "student");
    }

    match /classes/{classId} {
      allow read: if siteOpen() && hasProfile();
      allow create, update, delete: if siteOpen() && isAdmin();
    }

    match /subjects/{subjectId} {
      allow read: if siteOpen() && hasProfile();
      allow create, update, delete: if siteOpen() && isAdmin();
    }

    match /categories/{categoryId} {
      allow read: if siteOpen() && hasProfile();
      allow create, update, delete: if siteOpen() && isAdmin();
    }

    match /grades/{gradeId} {
      allow read: if siteOpen() &&
        (isAdmin() || (hasProfile() && resource.data.studentId == request.auth.uid));
      allow create, update, delete: if siteOpen() && isAdmin();
    }

    match /announcements/{announcementId} {
      allow read: if siteOpen() &&
        (isAdmin() || (hasProfile() && request.auth.uid in resource.data.recipientIds));
      allow create, update, delete: if siteOpen() && isAdmin();
    }

    match /events/{eventId} {
      allow read: if siteOpen() &&
        (isAdmin() || (hasProfile() && request.auth.uid in resource.data.recipientIds));
      allow create, update, delete: if siteOpen() && isAdmin();
    }

    match /savelists/{saveId} {
      allow read: if siteOpen() &&
        (isAdmin() || (canUseSaveList() && resource.data.ownerId == request.auth.uid));
      allow create: if canUseSaveList() && validNewSave();
      allow update: if siteOpen() &&
        (isAdmin() || (canUseSaveList() && validOwnSaveUpdate()));
      allow delete: if siteOpen() &&
        (isAdmin() || (canUseSaveList() && resource.data.ownerId == request.auth.uid));
    }
  }
}
